Usage Examples

Basic Navigation & Interaction

import browserPilot from 'browser-pilot';

// Connect to the browser
await browserPilot.connect();

// Navigate to a URL
await browserPilot.navigateToUrl('https://example.com');

// Click an element
await browserPilot.clickElement({ 
  selector: '#login-button' 
});

// Type text and submit
await browserPilot.typeText({
  selector: 'input[name="search"]',
  textToType: 'browser automation',
  submit: true
});

Screenshots & Storage

// Take a screenshot
const screenshot = await browserPilot.takeScreenshot({
  fullPage: true,
  saveToFile: true
});

// Access local storage
const storage = await browserPilot.getLocalStorage();
console.log('Stored data:', storage);

// Get cookies
const cookies = await browserPilot.getCookies();
console.log('Session cookies:', cookies);

Network & Console Monitoring

// Monitor network requests
const requests = await browserPilot.getNetworkRequests({
  method: 'POST',
  limit: 10
});

// Get console logs
const logs = await browserPilot.getConsoleLogs({
  level: 'error',
  limit: 50
});

// Get DOM snapshot
const dom = await browserPilot.getDomSnapshot({
  selector: '.main-content'
});
